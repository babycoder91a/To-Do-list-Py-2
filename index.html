# Initialize an empty list to store tasks # This list will hold all the tasks, with each task represented as a dictionary containing # a task description and its completion status (True for completed, False for pending) tasks = [] # Function to display the tasks def display_tasks(): # Check if there are any tasks in the list if len(tasks) == 0: print("\nNo tasks to show!") # If no tasks, display this message else: print("\nTo-Do List:") # If there are tasks, print a header for the list # Loop through all tasks in the list and display them for i, task in enumerate(tasks, start=1): # Start counting tasks from 1 status = "Completed" if task["completed"] else "Pending" # Check if the task is completed or not # Print the task number, task description, and its status print(f"{i}. {task['task']} - {status}") # Function to add a new task def add_task(): # Prompt the user to enter the task description task = input("Enter the task: ") # Append the new task to the tasks list as a dictionary with 'completed' set to False (not completed) tasks.append({"task": task, "completed": False}) print(f"Task '{task}' added to the list.") # Inform the user that the task was added # Function to mark a task as completed def mark_task_complete(): display_tasks() # Display the current tasks first so the user knows the options while True: try: # Ask the user to enter the number of the task they want to mark as complete task_num = int(input("Enter the task number to mark as complete: ")) # Check if the entered task number is valid (within the range of tasks available) if 1 <= task_num <= len(tasks): # Mark the selected task as completed by changing the 'completed' value to True tasks[task_num - 1]["completed"] = True print(f"Task {task_num} marked as complete.") # Inform the user that the task was marked break # Exit the loop after successful completion else: print("Invalid task number. Please enter a valid task number.") # Inform the user of invalid input except ValueError: print("Invalid input! Please enter a number.") # Handle the case where the user enters something that's not a number # Function to delete a task def delete_task(): display_tasks() # Display the current tasks before asking for input while True: try: # Ask the user to enter the task number they want to delete task_num = int(input("Enter the task number to delete: ")) # Check if the entered task number is valid (within the range of tasks available) if 1 <= task_num <= len(tasks): # Remove the selected task from the list deleted_task = tasks.pop(task_num - 1) print(f"Task '{deleted_task['task']}' deleted from the list.") # Inform the user that the task was deleted break # Exit the loop after successful deletion else: print("Invalid task number. Please enter a valid task number.") # Inform the user of invalid input except ValueError: print("Invalid input! Please enter a number.") # Handle the case where the user enters something that's not a number # Main function to run the program def main(): # Infinite loop to keep the program running until the user chooses to quit while True: # Show the menu options to the user print("\nOptions:") print("1. Add task") print("2. View tasks") print("3. Mark task as complete") print("4. Delete task") print("5. Quit") # Ask the user to choose an option choice = input("Choose an option (1-5): ") # Perform actions based on the user's choice if choice == "1": add_task() # Call the function to add a new task elif choice == "2": display_tasks() # Call the function to display the current list of tasks elif choice == "3": mark_task_complete() # Call the function to mark a task as completed elif choice == "4": delete_task() # Call the function to delete a task elif choice == "5": print("Exiting the program.") # Inform the user the program is exiting break # Exit the loop, ending the program else: print("Invalid option, please choose a valid option (1-5).") # Inform the user of invalid menu selection # Run the program if __name__ == "__main__": main() # Call the main function to start the program
